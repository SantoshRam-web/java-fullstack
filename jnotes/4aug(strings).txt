Q) Write a java program to find out number of digits ,find it sum and check it is even or odd?

input:
	NEX123

output;
	Number of Digits : 3
	Sum of Digits : 6
	It is even number 

class Test  
{
	public static void main(String[] args) 
	{
		String str="NEX123";
		
		String s = str.replaceAll("[^0-9]","");
		
		System.out.println("Number of Digits :"+s.length());
		
		int sum=0;
		for(int i=0;i<s.length();i++)
		{
			int n = Character.getNumericValue(s.charAt(i));
			sum+=n;
		}
		System.out.println("Sum of Digits :"+sum);
		
		if(sum%2==0)
			System.out.println("It is even number");
		else
			System.out.println("It is odd number");
	}
}

	{

Q) Write a java program to convert uppercase to lowercase and lowercase to uppercase letters?

input:
	IhUbTAlenT

output:
	iHuBtaLENt 

class Test  
{
	public static void main(String[] args) 
	{
		String str = "IhUbTAlenT";
		
		char[] carr = str.toCharArray();
		
		for(int i=0;i<str.length();i++)
		{
			if(carr[i]>='A' && carr[i]<='Z')
				carr[i] = (char)(carr[i]+32);
			else if(carr[i]>='a' && carr[i]<='z')
				carr[i] = (char)(carr[i]-32);
		}
		str="";
		for(char c : carr)
		{
			str+=c;
		}
		System.out.println(str);
	}
}


Q) Write a java program to display the string in a given format?

input:
	XYZ

	XY 
	XZ 
	YX 
	YZ 
	ZX 
	ZY 

ex:
--
class Test  
{
	public static void main(String[] args) 
	{
		String str = "XYZ";
		
		for(int i=0;i<str.length();i++)
		{
			for(int j=0;j<str.length();j++)
			{
				if(i!=j)
				{
					System.out.println(str.charAt(i)+""+str.charAt(j));
				}
			}
		}
	}
}

Q) Write a java program to display duplicate and unique characters from given string?

input:
	sessiseppi 

output:
	duplicates : sepi 
	uniques   : seip


class Test  
{
	public static void main(String[] args) 
	{
		String str = "sessiseppi";
		
		String duplicates="";
		String unique="";
		
		for(int i=0;i<str.length();i++)
		{
			String current = Character.toString(str.charAt(i));
			
			if(unique.contains(current))
			{
				if(!duplicates.contains(current))
				{
					duplicates+=current;
					continue;
				}
				else
				{
					continue;
				}
			}
			unique+=current;
		}
		System.out.println("Duplicate characters :"+duplicates);
		System.out.println("Unique Characters :"+unique);
	}
}
